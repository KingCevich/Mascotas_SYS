# Librería de machine learning. Aquí vive el modelo MobileNetV2
# que se usa para "leer" la foto de la mascota y convertirla en un
# vector de 1280 dimensiones que representa sus características
# visuales.
tensorflow==2.21.0

# API de alto nivel para construir y usar modelos de deep learning.
# Viene integrada con TensorFlow y es la que realmente carga y
# ejecuta MobileNetV2.
keras==3.14.1
    
# Librería de cálculo numérico. Se usa para manipular los vectores
# de la imagen (1280 números) y calcular la similitud coseno entre
# dos fotos distintas.
numpy==2.4.6
    
# Librería de procesamiento de imágenes (abrir, redimensionar,
# convertir a RGB) antes de pasarle la foto al modelo de IA.
pillow==12.2.0
    
# Framework que expone el microservicio de IA como una API REST
# (el endpoint que mascotas_serv llama vía Celery para analizar
# una foto).
fastapi==0.137.1
    
# Servidor ASGI que ejecuta la aplicación FastAPI del microservicio
# de IA.
uvicorn==0.49.0

# Necesaria para que FastAPI pueda recibir archivos (la foto de la
# mascota) en peticiones multipart/form-data.
python-multipart==0.0.32
    
# Usada para hacer peticiones HTTP hacia otros microservicios
# (ej: avisar a mascotas_serv o notificaciones_serv del resultado).
requests==2.34.2
